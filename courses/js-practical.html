JavaScript简介

JavaScript是一种高级的、解释型的编程语言，最初由Netscape公司的Brendan Eich在1995年创建。它被设计为在浏览器中运行，为网页添加交互功能。如今，JavaScript已成为Web开发的三大核心技术之一（与HTML和CSS并列），并且通过Node.js等技术，JavaScript也可以用于服务器端开发。

JavaScript是动态类型、基于原型的多范式语言，支持面向对象、命令式和函数式编程风格。随着ECMAScript标准的发展，JavaScript不断引入新特性，使其成为功能强大且灵活的编程语言。

JavaScript基础语法

JavaScript的基础语法是学习这门语言的第一步。了解变量、数据类型、运算符和控制结构是编写JavaScript程序的基础。

// 变量声明 let name = "赛博朋克"; // let声明块级作用域变量 const PI = 3.14159; // const声明常量 var oldVar = "旧式声明"; // var声明函数作用域变量（不推荐） // 数据类型 let number = 42; // 数字 let string = "Hello, World!"; // 字符串 let boolean = true; // 布尔值 let array = [1, 2, 3, 4, 5]; // 数组 let object = { // 对象 name: "YOCIM", age: 25, isDeveloper: true }; let nullValue = null; // null let undefinedValue; // undefined // 运算符 let sum = 10 + 5; // 加法 let difference = 10 - 5; // 减法 let product = 10 * 5; // 乘法 let quotient = 10 / 5; // 除法 let remainder = 10 % 3; // 取模 let isEqual = 10 === 10; // 严格相等 let isNotEqual = 10 !== 5; // 不相等 // 控制结构 if (age >= 18) { console.log("成年人"); } else { console.log("未成年人"); } // 循环 for (let i = 0; i < 5; i++) { console.log("迭代次数: " + i); } let i = 0; while (i < 5) { console.log("While循环: " + i); i++; }

JavaScript基础要点：

  • 变量声明：使用let和const，避免使用var
  • 数据类型：JavaScript有7种原始数据类型和对象类型
  • 类型转换：JavaScript会自动进行类型转换（隐式转换）
  • 严格模式：使用'use strict'开启严格模式，避免常见错误
  • 注释：使用//进行单行注释，/* */进行多行注释

DOM操作

DOM（文档对象模型）是HTML和XML文档的编程接口。它提供了对文档的结构化表示，并定义了一种方式，使程序可以访问和修改文档的内容、结构和样式。JavaScript通过DOM API与网页进行交互。

// 选择DOM元素 const elementById = document.getElementById("myId"); const elementsByClass = document.getElementsByClassName("myClass"); const elementsByTag = document.getElementsByTagName("div"); const elementBySelector = document.querySelector("#myId .myClass"); const elementsBySelectorAll = document.querySelectorAll(".myClass"); // 创建和添加元素 const newDiv = document.createElement("div"); newDiv.textContent = "新创建的div元素"; newDiv.className = "dom-element"; newDiv.id = "newDiv"; document.body.appendChild(newDiv); // 修改元素 const myElement = document.getElementById("myElement"); myElement.textContent = "修改后的文本"; myElement.innerHTML = "HTML内容"; myElement.style.color = "#00f3ff"; myElement.style.backgroundColor = "#0a0a14"; myElement.classList.add("active"); myElement.classList.remove("inactive"); myElement.classList.toggle("highlight"); // 删除元素 const elementToRemove = document.getElementById("removeMe"); elementToRemove.parentNode.removeChild(elementToRemove); // 或使用现代方法 elementToRemove.remove(); // 遍历DOM const parent = document.getElementById("parent"); const children = parent.children; const firstChild = parent.firstElementChild; const lastChild = parent.lastElementChild; const nextSibling = element.nextElementSibling; const prevSibling = element.previousElementSibling;

事件处理

事件是用户在网页上进行的交互操作，如点击、鼠标移动、键盘输入等。JavaScript通过事件处理程序来响应这些事件，使网页具有交互性。

// 添加事件监听器 const button = document.getElementById("myButton"); // 方法1: 使用addEventListener button.addEventListener("click", function(event) { console.log("按钮被点击了!", event); event.preventDefault(); // 阻止默认行为 event.stopPropagation(); // 阻止事件冒泡 }); // 方法2: 使用内联事件处理 button.onclick = function() { console.log("按钮点击事件"); }; // 常见事件类型 element.addEventListener("click", handleClick); // 鼠标点击 element.addEventListener("dblclick", handleDblClick); // 鼠标双击 element.addEventListener("mouseover", handleMouseOver); // 鼠标悬停 element.addEventListener("mouseout", handleMouseOut); // 鼠标移出 element.addEventListener("mousemove", handleMouseMove); // 鼠标移动 element.addEventListener("keydown", handleKeyDown); // 按键按下 element.addEventListener("keyup", handleKeyUp); // 按键释放 element.addEventListener("submit", handleSubmit); // 表单提交 element.addEventListener("change", handleChange); // 值改变 element.addEventListener("load", handleLoad); // 资源加载完成 // 事件对象 function handleClick(event) { console.log("事件类型:", event.type); console.log("目标元素:", event.target); console.log("当前元素:", event.currentTarget); console.log("鼠标位置 - X:", event.clientX, "Y:", event.clientY); console.log("按键代码:", event.keyCode); } // 事件委托 document.getElementById("parent").addEventListener("click", function(event) { if (event.target.classList.contains("child")) { console.log("子元素被点击:", event.target); } }); // 移除事件监听器 function handleEvent() { console.log("事件处理"); } element.addEventListener("click", handleEvent); element.removeEventListener("click", handleEvent);

函数与作用域

函数是JavaScript中的一等公民，是执行特定任务的代码块。函数可以接受参数并返回值。作用域决定了变量的可见性和生命周期，是理解JavaScript的关键概念。

// 函数声明 function greet(name) { return "Hello, " + name + "!"; } // 函数表达式 const greet = function(name) { return "Hello, " + name + "!"; }; // 箭头函数 (ES6+) const greet = (name) => { return "Hello, " + name + "!"; }; // 简化的箭头函数 const greet = name => "Hello, " + name + "!"; // 默认参数 function greet(name = "开发者") { return "Hello, " + name + "!"; } // 剩余参数 function sum(...numbers) { return numbers.reduce((total, num) => total + num, 0); } // 作用域 let globalVar = "全局变量"; // 全局作用域 function outer() { let outerVar = "外部变量"; // 函数作用域 function inner() { let innerVar = "内部变量"; // 函数作用域 console.log(globalVar); // 可以访问 console.log(outerVar); // 可以访问 console.log(innerVar); // 可以访问 } inner(); // console.log(innerVar); // 错误: innerVar未定义 } // 闭包 function createCounter() { let count = 0; return function() { count++; return count; }; } const counter = createCounter(); console.log(counter()); // 1 console.log(counter()); // 2 console.log(counter()); // 3 // 立即执行函数表达式 (IIFE) (function() { console.log("IIFE立即执行"); })(); // 高阶函数 function higherOrderFunction(callback) { console.log("高阶函数开始"); callback(); console.log("高阶函数结束"); } higherOrderFunction(function() { console.log("回调函数执行"); });

作用域与闭包要点：

  • 全局作用域：在函数外部声明的变量，在整个程序中可访问
  • 函数作用域：在函数内部声明的变量，只能在函数内部访问
  • 块级作用域：使用let和const声明的变量具有块级作用域
  • 闭包：函数可以记住并访问其词法作用域，即使函数在其作用域外执行
  • 变量提升：var声明会被提升到作用域顶部，但赋值不会提升

异步编程

异步编程是JavaScript的核心特性之一，允许代码在等待长时间操作（如网络请求、文件读取等）完成的同时继续执行其他任务。JavaScript使用事件循环模型来处理异步操作。

// 回调函数（传统方式） function fetchData(callback) { setTimeout(() => { const data = { id: 1, name: "赛博朋克数据" }; callback(null, data); }, 1000); } fetchData(function(error, data) { if (error) { console.error("错误:", error); } else { console.log("数据:", data); } }); // Promise function fetchDataPromise() { return new Promise((resolve, reject) => { setTimeout(() => { const success = Math.random() > 0.3; if (success) { resolve({ id: 1, name: "Promise数据" }); } else { reject("获取数据失败"); } }, 1000); }); } fetchDataPromise() .then(data => { console.log("成功:", data); return data.id; }) .then(id => { console.log("ID:", id); }) .catch(error => { console.error("错误:", error); }) .finally(() => { console.log("操作完成"); }); // Promise链 function getUser(id) { return Promise.resolve({ id, name: "用户" + id }); } function getPosts(userId) { return Promise.resolve(["文章1", "文章2", "文章3"]); } getUser(1) .then(user => { console.log("用户:", user); return getPosts(user.id); }) .then(posts => { console.log("文章:", posts); }); // async/await (ES2017) async function fetchUserAndPosts() { try { const user = await getUser(1); console.log("用户:", user); const posts = await getPosts(user.id); console.log("文章:", posts); return { user, posts }; } catch (error) { console.error("错误:", error); } } fetchUserAndPosts(); // Promise.all - 并行执行 Promise.all([getUser(1), getUser(2), getUser(3)]) .then(users => { console.log("所有用户:", users); }); // Promise.race - 竞速 Promise.race([ new Promise(resolve => setTimeout(() => resolve("第一个"), 500)), new Promise(resolve => setTimeout(() => resolve("第二个"), 1000)) ]).then(result => { console.log("获胜者:", result); // "第一个" });

ES6+新特性

ECMAScript 2015 (ES6)及其后续版本为JavaScript引入了许多强大的新特性，使代码更简洁、更易读、更强大。这些特性已经成为现代JavaScript开发的标准。

// 解构赋值 const person = { name: "YOCIM", age: 25, city: "赛博城" }; const { name, age } = person; console.log(name, age); // YOCIM 25 const colors = ["红", "绿", "蓝"]; const [firstColor, secondColor] = colors; console.log(firstColor, secondColor); // 红 绿 // 模板字面量 const greeting = `Hello, ${name}! 你今年${age}岁，住在${person.city}。 2 + 2 = ${2 + 2}`; console.log(greeting); // 扩展运算符 const arr1 = [1, 2, 3]; const arr2 = [4, 5, 6]; const combined = [...arr1, ...arr2]; // [1, 2, 3, 4, 5, 6] const obj1 = { a: 1, b: 2 }; const obj2 = { c: 3, d: 4 }; const merged = { ...obj1, ...obj2 }; // { a: 1, b: 2, c: 3, d: 4 } // 对象字面量增强 const x = 10, y = 20; const point = { x, y }; // 等价于 { x: x, y: y } const methodName = "calculate"; const calculator = { [methodName]() { return this.x + this.y; } }; // 可选链操作符 (ES2020) const user = { profile: { name: "YOCIM", address: { city: "赛博城" } } }; const city = user?.profile?.address?.city; // "赛博城" const country = user?.profile?.address?.country; // undefined (不会报错) // 空值合并运算符 (ES2020) const settings = { theme: null, fontSize: 16 }; const theme = settings.theme ?? "dark"; // "dark" (只有null或undefined时使用默认值) const fontSize = settings.fontSize ?? 14; // 16 // 类 (ES6) class Animal { constructor(name) { this.name = name; } speak() { console.log(`${this.name} 发出声音`); } } class Dog extends Animal { constructor(name, breed) { super(name); this.breed = breed; } speak() { console.log(`${this.name} 汪汪叫`); } static info() { return "这是一个Dog类"; } } const myDog = new Dog("赛博狗", "机械犬"); myDog.speak(); // 赛博狗 汪汪叫 // 模块 (ES6) // module.js export const PI = 3.14159; export function circleArea(radius) { return PI * radius * radius; } export default class Circle { constructor(radius) { this.radius = radius; } } // app.js import Circle, { PI, circleArea } from './module.js'; const myCircle = new Circle(5); console.log(circleArea(myCircle.radius));

ES6+核心特性：

  • let/const：块级作用域变量声明
  • 箭头函数：简洁的函数语法和词法this绑定
  • 模板字面量：多行字符串和字符串插值
  • 解构赋值：从数组或对象中提取值
  • 扩展运算符：展开数组或对象
  • 类：基于类的面向对象编程语法
  • 模块：原生的模块系统
  • Promise：更好的异步处理方式
  • async/await：使异步代码看起来像同步代码

// 待办事项应用核心代码 class TodoApp { constructor() { this.todos = this.loadTodos(); this.render(); this.setupEventListeners(); } loadTodos() { const saved = localStorage.getItem('cyberpunk-todos'); return saved ? JSON.parse(saved) : []; } saveTodos() { localStorage.setItem('cyberpunk-todos', JSON.stringify(this.todos)); } addTodo(text) { if (!text.trim()) return; const todo = { id: Date.now(), text: text, completed: false, createdAt: new Date().toISOString() }; this.todos.push(todo); this.saveTodos(); this.render(); } toggleTodo(id) { this.todos = this.todos.map(todo => todo.id === id ? { ...todo, completed: !todo.completed } : todo ); this.saveTodos(); this.render(); } deleteTodo(id) { this.todos = this.todos.filter(todo => todo.id !== id); this.saveTodos(); this.render(); } clearCompleted() { this.todos = this.todos.filter(todo => !todo.completed); this.saveTodos(); this.render(); } clearAll() { this.todos = []; this.saveTodos(); this.render(); } render() { // 更新统计 const totalCount = this.todos.length; const completedCount = this.todos.filter(todo => todo.completed).length; const pendingCount = totalCount - completedCount; // 渲染列表 const todoList = document.getElementById('todoList'); todoList.innerHTML = ''; this.todos.forEach(todo => { const li = document.createElement('li'); li.className = `todo-item ${todo.completed ? 'completed' : ''}`; li.innerHTML = ` ${todo.text}
`; todoList.appendChild(li); }); // 更新统计显示 document.getElementById('totalCount').textContent = totalCount; document.getElementById('pendingCount').textContent = pendingCount; document.getElementById('completedCount').textContent = completedCount; } setupEventListeners() { document.getElementById('addTodo').addEventListener('click', () => { const input = document.getElementById('todoInput'); this.addTodo(input.value); input.value = ''; }); document.getElementById('todoInput').addEventListener('keypress', (e) => { if (e.key === 'Enter') { const input = document.getElementById('todoInput'); this.addTodo(input.value); input.value = ''; } }); document.getElementById('todoList').addEventListener('click', (e) => { if (e.target.classList.contains('toggle-btn')) { const id = parseInt(e.target.dataset.id); this.toggleTodo(id); } if (e.target.classList.contains('delete-btn')) { const id = parseInt(e.target.dataset.id); this.deleteTodo(id); } }); document.getElementById('clearCompleted').addEventListener('click', () => { this.clearCompleted(); }); document.getElementById('clearAllTodos').addEventListener('click', () => { if (confirm('确定要删除所有待办事项吗？')) { this.clearAll(); } }); } } // 初始化应用 const app = new TodoApp();
